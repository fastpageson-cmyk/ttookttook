# 데이터 모델 (프로토타입 State 정의)

```typescript
// 사용자 — 실제 로그인 없음. 시작 화면에서 닉네임만 입력.
interface AppUser {
  nickname: string
}

// 사전 진단 (화면기획서 3-3, 3-4)
interface DiagnosisState {
  answers: number[]        // 20문항 응답 (문항별 선택지 인덱스)
  score: number             // 0~100 환산 점수
  peerAverage: 62.6         // 고정값, 한은 2024 조사 20대 평균
  completedAt: string | null
}

// 0단계 시뮬레이션 세션 (화면기획서 3-5)
interface SimulationSession {
  status: "not_started" | "in_progress" | "ended"
  currentWeek: number        // mock-data의 week 인덱스, 1~520
  startingCash: number       // 초기 자금, 목업: 10,000,000
  cash: number
  holdings: Record<string, number>   // 종목코드 -> 보유수량, 예: { "A전자": 10 }
  trades: Trade[]
}

interface Trade {
  id: string
  week: number              // 체결 시점 (mock-data week)
  stockCode: string          // stocks.json의 code
  type: "buy" | "sell"
  quantity: number
  price: number               // 체결가 (mock-data 종가 기준)
  fee: number                  // 수수료+세금 목업 상수 적용
}

// 0단계 종료 리포트 (화면기획서 3-6) — SimulationSession 종료 시 계산해서 생성
interface SessionReport {
  finalReturn: number         // 수익률 (%)
  mdd: number                  // 최대낙폭 (%)
  totalTrades: number
  totalFees: number
  benchmarkReturn: number      // 같은 자금을 지수ETF에 넣었을 때 수익률
}

// AI 매매 실수 리포트 (화면기획서 3-7)
// 주의: 이번 프로토타입 단계에서는 실제 LLM 호출이 아니라 규칙 기반으로 구현 권장.
// 이유는 [[빌드_브리프_클로드코드용]] "AI 기능 처리 방침" 참고.
interface MistakeReportItem {
  tradeId: string
  headline: string    // 예: "가장 크게 손해 본 매매"
  explanation: string  // 템플릿 기반 자연어 설명
}
interface AIMistakeReport {
  items: MistakeReportItem[]  // 3건
}

// 1주차 학습 진행 상태 (화면기획서 3-9~3-11)
interface Week1Progress {
  cardsViewed: number[]       // 열람한 강의 카드 번호(1~10)
  practicesCompleted: {
    review0: boolean          // 실습1: 0단계 복기
    rateCutReplay: boolean    // 실습2: 금리 인하기 리플레이
    fxWidget: boolean         // 실습3: 한미 금리차·환율 위젯
  }
  quiz: {
    attempts: number
    lastScore: number         // 5문항 중 정답 수
    passed: boolean            // 통과 기준: 팀 확인 필요, 잠정 4/5 이상
  }
}

// 앱 전체 state (위 조각들을 묶는 최상위)
interface AppState {
  user: AppUser
  diagnosis: DiagnosisState
  simulation: SimulationSession
  aiReport: AIMistakeReport | null
  week1: Week1Progress
}
```

## 참고 사항

- **퀴즈 통과 기준(5문항 중 몇 개 이상)**은 아직 정해지지 않았습니다. 위에서는 잠정적으로 4/5로 표시했으니 팀에서 확정해주세요.
- **저장 방식**은 정하지 않았습니다(브라우저 메모리만 쓸지, 새로고침해도 유지되게 로컬 저장을 쓸지는 Claude Code 구현 단계에서 판단해도 되는 부분입니다).
- 이 구조는 0단계 + 1주차(MVP)까지만 다룹니다. 2주차 이후 데이터 모델은 해당 단계 착수 시 추가해야 합니다.
